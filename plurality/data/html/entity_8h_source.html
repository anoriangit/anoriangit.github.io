50  // serialize and deserialize
51  MessagePtr serialize(uint64_t entityid) {
52  int payload_size = m_pb->ByteSize();
53  auto m = MessageBuffer::Create(payload_size);
54 
55  m->Header()->SetOP(OPCODE::RPCREQUEST);
56  m->Header()->SetENTITYID(entityid);
57 
58  bool result = m_pb->SerializeToArray(m->Payload(), payload_size);
59  assert(result);
60  return m;
61  }

114  // NOTE: ABOUT THE ENTITY ID FIELD
115  // The entity ID is a 64 bit uint64_t referred to as the "raw" ID
116  // - the lowest 12 bits encode the entity type (0-4095)
117  // - the 12 bits following that are reserved for flag bits
118  // - the upper 40 bits are the actual "pure" ID

120  // and a 32 bits ID in the future
